
@layout MainLayout
@implements IAsyncDisposable
@inject IJSRuntime JS
@inject OrderTimelineService _TimelineService
@inject NavigationManager _NavigationManager
@inject AuthenticationStateProvider AuthStateProvider
@attribute [Authorize(Roles = "Admin,Supervisor,Operator")]

<PageTitle>Detalle de pedido — Tommy Logistic</PageTitle>

@if (_loading)
{
    <div class="loading-overlay">
        <div class="spinner"></div>
        <span>Cargando pedido...</span>
    </div>
}
else if (_timeline is null)
{
    <div class="empty-state">
        <span class="empty-icon">📭</span>
        <p class="text-bold">Pedido no encontrado</p>
        <button class="btn btn-ghost mt-4" @onclick="@(() => _NavigationManager.NavigateTo("/Orders"))">← Volver</button>
    </div>
}
else
{
    <div class="od-breadcrumb">
        <button class="btn btn-ghost btn-sm" @onclick="GoBack">← Atrás</button>
        <span class="text-muted text-sm">/</span>
        <span class="text-sm">Pedido</span>
        <span class="text-muted text-sm">/</span>
        <span class="od-tracking">@_timeline.TrackingCode</span>
        <span class="od-realtime-dot @(_connected ? "od-realtime-dot--on" : "od-realtime-dot--off")">
            @(_connected ? "● En vivo" : "○ Reconectando...")
        </span>
    </div>

    <div class="od-layout">

        <div class="od-info-col">
            <div class="card od-main-card">
                <div class="od-status-row">
                    <BadgeEstado Estado="@_timeline.CurrentStatus" />
                    @if (_timeline.TotalAttempts > 0)
                    {
                        <span class="od-attempts">
                            @_timeline.TotalAttempts intento@(_timeline.TotalAttempts > 1 ? "s" : "")
                        </span>
                    }
                </div>
                <h1 class="od-recipient">@_timeline.RecipientName</h1>
                <p class="od-address text-muted text-sm">📍 @_timeline.Address</p>
                <p class="od-code text-sm">
                    <span class="codigo-seguimiento">@_timeline.TrackingCode</span>
                </p>
            </div>

            <div class="card">
                <div class="card-header">
                    <span class="card-title">Acciones</span>
                </div>
                <div class="od-actions">
                    @if (CanPickup)
                    {
                        <button class="btn btn-primary w-full" @onclick="ShowPickupModal">
                            🛵 Registrar retiro
                        </button>
                    }
                    @if (CanDeliver)
                    {
                        <button class="btn btn-success w-full" @onclick="ShowDeliveredModal">
                            ✅ Confirmar entrega
                        </button>
                        <button class="btn btn-warning w-full" @onclick="ShowAbsentModal">
                            🚪 Destinatario ausente
                        </button>
                    }
                    @if (CanStoreBaglok)
                    {
                        <button class="btn btn-outline w-full" @onclick="ShowBaglokModal">
                            📦 Guardar en Baglok
                        </button>
                    }
                    @if (CanReschedule)
                    {
                        <button class="btn btn-ghost w-full" @onclick="ShowRescheduleModal">
                            📅 Reprogramar
                        </button>
                    }
                    @if (CanRelease)
                    {
                        <button class="btn btn-primary w-full" @onclick="ShowReleaseModal">
                            🔓 Liberar del Baglok
                        </button>
                    }
                    <button class="btn btn-ghost w-full" @onclick="ShowNoteModal">
                        📝 Agregar nota
                    </button>
                </div>
            </div>
        </div>

        <div class="od-timeline-col">
            <div class="card">
                <div class="card-header">
                    <span class="card-title">Trazabilidad del pedido</span>
                    <span class="text-xs text-muted">
                        @_timeline.Events.Count evento@(_timeline.Events.Count != 1 ? "s" : "")
                    </span>
                </div>

                @if (_showNewEventToast)
                {
                    <div class="od-toast">
                        <span>🔔</span>
                        <span>Nuevo evento registrado en tiempo real</span>
                    </div>
                }

                <OrderTimeline Events="@_timeline.Events" />
            </div>
        </div>
    </div>

    <!-- Modales -->
    @if (_showPickupModal)
    {
        <div class="od-modal-overlay" @onclick="CloseModals">
            <div class="od-modal" @onclick:stopPropagation="true">
                <h3 class="od-modal-title">🛵 Registrar retiro</h3>
                <p class="text-sm text-muted">¿Confirmas que el repartidor retiró este pedido?</p>
                <div class="od-modal-actions">
                    <button class="btn btn-ghost" @onclick="CloseModals">Cancelar</button>
                    <button class="btn btn-primary" @onclick="ConfirmPickup" disabled="@_saving">
                        @(_saving ? "Guardando..." : "Confirmar")
                    </button>
                </div>
            </div>
        </div>
    }

    @if (_showBaglokModal)
    {
        <div class="od-modal-overlay" @onclick="CloseModals">
            <div class="od-modal" @onclick:stopPropagation="true">
                <h3 class="od-modal-title">📦 Guardar en Baglok</h3>
                <div class="form-group">
                    <label class="form-label required">Ubicación en Baglok</label>
                    <input class="form-control" placeholder="ej: ZONA-B / ESTANTE-3 / POS-12"
                           @bind="_baglokLocation" />
                </div>
                <div class="form-group">
                    <label class="form-label">Observación</label>
                    <textarea class="form-control" rows="2" @bind="_actionNote"></textarea>
                </div>
                <div class="od-modal-actions">
                    <button class="btn btn-ghost" @onclick="CloseModals">Cancelar</button>
                    <button class="btn btn-primary" @onclick="ConfirmBaglok"
                            disabled="@(string.IsNullOrWhiteSpace(_baglokLocation) || _saving)">
                        @(_saving ? "Guardando..." : "Guardar en Baglok")
                    </button>
                </div>
            </div>
        </div>
    }

    @if (_showRescheduleModal)
    {
        <div class="od-modal-overlay" @onclick="CloseModals">
            <div class="od-modal" @onclick:stopPropagation="true">
                <h3 class="od-modal-title">📅 Reprogramar pedido</h3>
                <div class="form-group">
                    <label class="form-label required">Nueva fecha de entrega</label>
                    <input class="form-control" type="date" @bind="_rescheduledFor"
                           min="@DateTime.Today.AddDays(1).ToString("yyyy-MM-dd")" />
                </div>
                <div class="form-group">
                    <label class="form-label">Motivo</label>
                    <textarea class="form-control" rows="2" @bind="_actionNote"
                              placeholder="Razón de la reprogramación..."></textarea>
                </div>
                <div class="od-modal-actions">
                    <button class="btn btn-ghost" @onclick="CloseModals">Cancelar</button>
                    <button class="btn btn-warning" @onclick="ConfirmReschedule" disabled="@_saving">
                        @(_saving ? "Guardando..." : "Reprogramar")
                    </button>
                </div>
            </div>
        </div>
    }

    @if (_showAbsentModal)
    {
        <div class="od-modal-overlay" @onclick="CloseModals">
            <div class="od-modal" @onclick:stopPropagation="true">
                <h3 class="od-modal-title">🚪 Destinatario ausente</h3>
                <div class="form-group">
                    <label class="form-label">Observación</label>
                    <textarea class="form-control" rows="2" @bind="_actionNote"
                              placeholder="¿Qué ocurrió en el intento?"></textarea>
                </div>
                <div class="od-modal-actions">
                    <button class="btn btn-ghost" @onclick="CloseModals">Cancelar</button>
                    <button class="btn btn-warning" @onclick="ConfirmAbsent" disabled="@_saving">
                        @(_saving ? "Guardando..." : "Registrar ausencia")
                    </button>
                </div>
            </div>
        </div>
    }

    @if (_showNoteModal)
    {
        <div class="od-modal-overlay" @onclick="CloseModals">
            <div class="od-modal" @onclick:stopPropagation="true">
                <h3 class="od-modal-title">📝 Agregar nota</h3>
                <div class="form-group">
                    <label class="form-label required">Nota</label>
                    <textarea class="form-control" rows="3" @bind="_actionNote"
                              placeholder="Escribe una observación..."></textarea>
                </div>
                <div class="od-modal-actions">
                    <button class="btn btn-ghost" @onclick="CloseModals">Cancelar</button>
                    <button class="btn btn-primary" @onclick="ConfirmNote"
                            disabled="@(string.IsNullOrWhiteSpace(_actionNote) || _saving)">
                        @(_saving ? "Guardando..." : "Guardar nota")
                    </button>
                </div>
            </div>
        </div>
    }
}
